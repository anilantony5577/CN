#include <stdio.h>
#define INFINITY 9999
#define MAX 10
// Function to implement Dijkstra's algorithm
void dijkstra(int graph[MAX][MAX], int n, int start) {
    int distance[MAX], visited[MAX] = {0};
    int count, minDist, nextNode, i, j;

    // Step 1: Initialize distances from start node
    for (i = 0; i < n; i++) {
        distance[i] = graph[start][i];
    }
    visited[start] = 1;
    distance[start] = 0;
    // Step 2: Repeat for (n - 1) nodes
    for (count = 1; count < n - 1; count++) {
        minDist = INFINITY;
        // Step 3: Find the unvisited node with the smallest distance
        for (i = 0; i < n; i++) {
            if (distance[i] < minDist && !visited[i]) {
                minDist = distance[i];
                nextNode = i;
            }
        }
        visited[nextNode] = 1;
        // Step 4: Update distances to neighbors of the chosen node
        for (i = 0; i < n; i++) {
            if (!visited[i]) {
                if (minDist + graph[nextNode][i] < distance[i]) {
                    distance[i] = minDist + graph[nextNode][i];
                }
            }
        }
    }
    // Step 5: Print the shortest distances
    printf("\nShortest distances from node %d:\n", start);
    for (i = 0; i < n; i++) {
        printf("To node %d: %d\n", i, distance[i]);
    }
}
int main() {
    int graph[MAX][MAX], i, j, n, start;
    // Input number of nodes
    printf("Enter the number of nodes: ");
    scanf("%d", &n);
    // Input adjacency matrix
    printf("Enter the adjacency matrix (use 9999 for no direct path):\n");
    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++) {
            scanf("%d", &graph[i][j]);
        }
    }
    // Input starting node
    printf("Enter the starting node (0 to %d): ", n - 1);
    scanf("%d", &start);
    // Call Dijkstra's algorithm
    dijkstra(graph, n, start);
    return 0;
}
