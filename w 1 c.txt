#include <stdio.h> 
#include <string.h> 
int main() { 
int a[100], b[200]; 
int i, j = 0, k, count = 1, n; 
printf("Enter frame length: "); 
scanf("%d", &n); 
printf("Enter input frame (0's & 1's only):\n"); 
for(i = 0; i < n; i++) { 
        scanf("%d", &a[i]); 
    } 
    b[j++] = a[0];  // Copy first bit 
    for(i = 1; i < n; i++) { 
        if(a[i] == 1 && a[i-1] == 1) { 
            count++; 
        } else { 
            count = 1; 
        } 
        b[j++] = a[i]; 
        if(count == 5) { 
            b[j++] = 0;  // Stuff a 0 after five 1s 
            count = 0;   // Reset count after stuffing 
        }         } 
    printf("After bit stuffing, the frame is:\n"); 
    for(i = 0; i < j; i++) { 
        printf("%d", b[i]); 
    } 
    printf("\n"); 
    return 0; 
}
